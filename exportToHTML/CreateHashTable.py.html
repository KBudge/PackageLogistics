6    # HashTable class using chaining.
7    class ChainingHashTable:
8        # Constructor with optional initial capacity parameter.
9        # Assigns all indexes with an empty list.
10       def __init__(self, initial_capacity=10):
11           # initialize the hash table with empty index list entries.
12           self.table = []
13           for i in range(initial_capacity):
14               self.table.append([])
15   
16       # Inserts new Info into the hash table.
17       def insert(self, ID, Info): # does both insert and update
18           # get the index list where this Info will go.
19           index = hash(ID) % len(self.table)
20           index_list = self.table[index]
21   
22           # update Info if ID is already in the index
23           for k in index_list:
24               #print (Info)
25               if k[0] == ID:
26                   k[1] = Info
27                   return True
28   
29           # if not, insert the Info to the end of the index list.
30           key_value = [ID, Info]
31           index_list.append(key_value)
32           return True
33   
34       # Searches for Info with matching ID in the hash table.
35       # Returns the Info if found, or None if not found.
36       def search(self, ID):
37           # get the index list where this ID would be.
38           index = hash(ID) % len(self.table)
39           index_list = self.table[index]
40           #print(index_list)
41   
42           # search for the key in the bucket list
43           for k in index_list:
44               #print (Info)
45               if k[0] == ID:
46                   return k[1] #Info
47   
48           return None
49   
50       # Removes Info with matching ID from the hash table.
51       def remove(self, ID):
52           # get the index list where this Info will be removed from.
53           index = hash(ID) % len(self.table)
54           index_list = self.table[index]
55   
56           # remove the Info from the index list if it is present.
57           for k in index_list:
58               #print (key_value)
59               if k[0] == ID:
60                   index_list.remove([k[0],k[1]])
