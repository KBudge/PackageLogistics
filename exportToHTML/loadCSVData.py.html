20   # Function to read package CSV file and insert objects into hash table
21   def loadPackageData(fileName):
22       with open(fileName) as packageData:     # Read CSV file
23           reader = csv.reader(packageData)
24           for row in reader:          # Put each piece of data in the appropriate attribute of package
25               packageID = int(row[0])
26               address = row[1]
27               city = row[2]
28               state = row[3]
29               zip = row[4]
30               deadline = row[5]
31               weight = row[6]
32               notes = row[7]
33               loadTime = None         # No packages are loaded on any trucks
34               truck = None            # No packages are loaded on any trucks
35               status = 'HUB'          # All packages start at the hub
36               deliveryTime = None     # No packages are delivered yet
37   
38               # Create a package object
39               package = Package(packageID, address, city, state, zip, deadline, weight, notes, loadTime, truck, status, deliveryTime)
40               #print(package)
41   
42               # Insert package object into hash table
43               packageHash.insert(packageID, package)
44   
45   # Function to read distance data and put it in a list of lists
46   def loadDistanceData(fileName):
47       with open(fileName) as csvFile:     # Read CSV file
48           reader = csv.reader(csvFile)
49           for row in reader:
50               distanceData.append(row)    # Add distanceData to list
51   
52           # Fill in the whole distance table so distance from a to b is same as b to a
53           for x in range(len(distanceData)):
54               for y in range(len(distanceData)):
55                   distanceData[x][y] = distanceData[y][x]
56
